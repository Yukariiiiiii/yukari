从零搭建个人博客系统

2026-06-20 968 阅读 约 18 分钟
全栈 教程 Node.js

为什么要自己搭博客？

Hexo、Hugo、WordPress……现成方案一大把，为什么还要从零搭建？三个理由：完全可控、练手全栈、秀出个性。本文带你用 Node.js + 轻量前端，从空文件夹到线上博客。

技术选型：Node.js + Express（后端）+ SQLite（数据库）+ 原生 HTML/CSS/JS（前端）。零依赖前端框架，轻量、快、好理解。

一、项目结构

my-blog/
├── server.js          # 入口
├── db.js              # 数据库连接
├── routes/
│   ├── posts.js       # 文章 CRUD
│   └── auth.js        # 管理员登录
├── views/             # 服务端渲染模板
│   ├── layout.html
│   ├── home.html
│   └── post.html

│   ├── styles.css

├── content/           # Markdown 文章
│   └── hello-world.md
└── package.json

二、初始化项目

mkdir my-blog && cd my-blog
npm init -y
npm install express better-sqlite3 marked express-session bcrypt

三、数据库层

// db.js
const Database = require('better-sqlite3');
const db = new Database('blog.db');

db.exec(`
    CREATE TABLE IF NOT EXISTS posts (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        slug TEXT UNIQUE NOT NULL,
        title TEXT NOT NULL,
        content TEXT NOT NULL,
        excerpt TEXT,
        created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
        updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
    );

    CREATE TABLE IF NOT EXISTS users (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        username TEXT UNIQUE NOT NULL,
        password_hash TEXT NOT NULL
    );
`);

module.exports = db;

四、服务端核心

// server.js
const express = require('express');
const session = require('express-session');
const { marked } = require('marked');
const db = require('./db');
const postsRouter = require('./routes/posts');

const app = express();

app.set('view engine', 'html');
app.engine('html', require('express-file-engine'));

app.use(express.urlencoded({ extended: true }));
app.use(express.json());
app.use(express.static('public'));
app.use(session({
    secret: 'your-secret-key',
    resave: false,
    saveUninitialized: false
}));

app.use((req, res, next) => {
    res.locals.marked = marked;
    next();
});

app.use('/api/posts', postsRouter);

app.get('/', (req, res) => {
    const posts = db.prepare('SELECT * FROM posts ORDER BY created_at DESC').all();
    res.render('home', { posts, user: req.session.user });
});

app.get('/post/:slug', (req, res) => {
    const post = db.prepare('SELECT * FROM posts WHERE slug = ?').get(req.params.slug);
    if (!post) return res.status(404).send('Not Found');
    post.html = marked(post.content);
    res.render('post', { post, user: req.session.user });
});

app.listen(3000, () => console.log('Blog running at http://localhost:3000'));

五、文章 CRUD 路由

// routes/posts.js
const router = require('express').Router();
const db = require('../db');
const bcrypt = require('bcrypt');

router.post('/', requireAuth, (req, res) => {
    const { title, content, slug } = req.body;
    const excerpt = content.slice(0, 150) + '...';
    db.prepare('INSERT INTO posts (slug, title, content, excerpt) VALUES (?, ?, ?, ?)')
      .run(slug || title.toLowerCase().replace(/\s+/g, '-'), title, content, excerpt);
    res.redirect(`/post/${slug}`);
});

router.put('/:id', requireAuth, (req, res) => {
    const { title, content } = req.body;
    db.prepare('UPDATE posts SET title=?, content=?, updated_at=CURRENT_TIMESTAMP WHERE id=?')
      .run(title, content, req.params.id);
    res.json({ ok: true });
});

router.delete('/:id', requireAuth, (req, res) => {
    db.prepare('DELETE FROM posts WHERE id=?').run(req.params.id);
    res.json({ ok: true });
});

function requireAuth(req, res, next) {
    if (!req.session.user) return res.status(401).json({ error: 'Unauthorized' });
    next();
}

module.exports = router;

七、Markdown 写作体验

文章用 Markdown 写，发布时转 HTML。本地写作流程：

# 新文章
在 content/ 下放 .md 文件，格式：
---
title: 我的第一篇博客
slug: my-first-post
date: 2026-07-20
tags: [生活, 随笔]
---

正文用标准 Markdown 语法...

写一个导入脚本，把 content/ 下的 Markdown 批量导入数据库：

// scripts/import.js
const fs = require('fs');
const path = require('path');
const db = require('./db');
const { marked } = require('marked');
const fm = require('front-matter');

const dir = path.join(__dirname, '../content');
fs.readdirSync(dir).forEach(file => {
    if (!file.endsWith('.md')) return;
    const raw = fs.readFileSync(path.join(dir, file), 'utf8');
    const { attributes, body } = fm(raw);
    const slug = attributes.slug || file.replace('.md', '');
    const excerpt = body.slice(0, 150) + '...';
    db.prepare(`INSERT OR REPLACE INTO posts (slug, title, content, excerpt) VALUES (?, ?, ?, ?)`)
      .run(slug, attributes.title, body, excerpt);
    console.log(`✅ Imported: ${attributes.title}`);
});

八、部署上线

方案 A：Railway（最简单）

  1. GitHub 推代码
  2. Railway 关联仓库 → 自动检测 Node.js
  3. 添加 Volume 持久化 blog.db
  4. 自动获得 HTTPS 域名

方案 B：VPS + PM2

ssh root@your-server
git clone your-repo && cd your-repo
npm install && npm run build
npm install -g pm2
pm2 start server.js --name blog
pm2 save && pm2 startup

再配 Nginx 反代 + Let's Encrypt 证书即可。

方案 C：Serverless（Vercel / Cloudflare Pages）

把 Express 换成 Vercel 的 Serverless 函数格式，数据库改用 Turso（SQLite 的 Serverless 版）：

// api/posts.js → Vercel Serverless
import { Client } from '@libsql/client';

const db = createClient({
    url: process.env.TURSO_URL,
    authToken: process.env.TURSO_TOKEN
});

export default async function handler(req, res) {
    const posts = await db.execute('SELECT * FROM posts ORDER BY created_at DESC');
    res.json(posts.rows);
}
安全提醒：SQLite 文件不要暴露到公网；密码用 bcrypt 哈希存储；登录态用 HttpOnly Cookie + CSRF 防护。

小结

从零搭博客听起来吓人，其实核心就三件事：数据怎么存（SQLite）、请求怎么处理（Express）、页面怎么渲染（模板 + CSS）。搭好骨架后，每加一个功能都是一次完整的全栈练习。而且——拥有完全属于自己的小天地，那种感觉真的很棒。
